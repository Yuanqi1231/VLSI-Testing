\begin{frame}{为什么关注 D2D 互连的在役健康}

    \resizebox{\textwidth}{!}{%
    \begin{tabular}{@{}lllll@{}}
        \toprule
        \textbf{类别} & \textbf{改变的参数} & \textbf{故障取值（连续）} & \textbf{注入位置} & \textbf{机理} \\
        \midrule
        healthy & 无注入 & —— & —— & —— \\
        resistive\_open & 单段串阻 $+\,r_o$ & $r_o\sim\log U(20,500)\,\Omega$ & 段 $U\{2..17\}$ & 裂纹/弱开路 \\
        bump\_void\_aging & 单段串阻 $+\,r_o$（缓增） & $r_o\sim\log U(5,80)\,\Omega$ & 端段 $\{1,2,17,18\}$ & 凸点空洞老化 \\

    \vspace{0.4em}
    \begin{block}{\small 什么是``漂移（drift）故障''}
        \small 老化机理造成的\hl{缓变参数漂移}（r\_drift / c\_drift / bump\_void\_aging 的``缓增''）：
        无离散失效事件、严重度连续、\hl{今天还不误码}——诊断目标是在 BER 恶化之前量化它
    \end{block}
    \vspace{2pt}
    {\scriptsize 严重度均为连续采样（电阻类取 $\log U$，效应是乘性的）；温度 $T\sim U(27,110)^\circ$C 对所有类同分布}

\begin{frame}{眼图征象 1/5：衰减类——串联电阻的四种来源}
    \centering
    \includegraphics[width=0.8\textwidth]{pic/eye_grp_attenuation.png}
    \begin{itemize}\footnotesize
        \item 根因同为\hl{串阻分压}（对 $R_\text{term}$）+ RC 损耗 $\Rightarrow$ 眼高塌陷（354 $\to$ 103--209 mV）
        \item 区别在\hl{位置与分布}：单段集中（开路）/ 凸点端缓增（空洞老化）/ 5 段局部 + 热噪声（hotspot）/ 全线均摊（r\_drift，漂移故障）——这也是它们相互混淆的物理根源
    \end{itemize}
\end{frame}

\begin{frame}{眼图征象 2/5：摆幅类——supply\_droop}
    \centering
    \includegraphics[width=0.62\textwidth]{pic/eye_grp_droop.png}
    \begin{itemize}\footnotesize
        \item 根因在\hl{发送端}：launch 摆幅 $\times(1-d)$，信道本身完好 $\Rightarrow$ 眼形干净、整体变矮（354 $\to$ 292 mV），无额外抖动/噪声
        \item 与\hl{良性负载 droop 完全同机制}，只是幅度更大且\hl{未被遥测上报}——模型必须发现``摆幅比 droop monitor 所解释的更小''，是遥测融合实验的关键类
    \end{itemize}
\end{frame}

\begin{frame}{眼图征象 3/5：边沿类——c\_drift（漂移故障）}
    \centering
    \includegraphics[width=0.62\textwidth]{pic/eye_grp_cdrift.png}
    \begin{itemize}\footnotesize
        \item 漂移故障之二：介质退化/吸湿使\hl{全部节点电容}缓慢上涨 $\times(1{+}c)$
        \item RC 时间常数增大 $\Rightarrow$ \hl{边沿变缓 + ISI}：眼宽从 51 ps 坍到 22 ps、BER 升至 $10^{-7}$，而幅度损失相对次要——与衰减类的``矮眼''形成正交征象
        \item 严重度连续 $\Rightarrow$ 可回归（$R^2=0.908$）：在误码爆发前给出退化程度
    \end{itemize}
\end{frame}

\begin{frame}{眼图征象 4/5：串扰/桥接类——来自邻线的能量}
    \centering
    \includegraphics[width=0.8\textwidth]{pic/eye_grp_xtalk.png}
    \begin{itemize}\footnotesize
        \item crosstalk\_cap：全部节点耦合容 $\times f$ $\Rightarrow$ 攻击线码型经 $C_c$ 注入，噪声\hl{与受害码型无关}——整个眼均匀模糊
        \item bridge\_short：受害--邻线同号节点间的\hl{电阻泄漏通道}，双向分压 + 电平随邻线数据偏移——眼分层、出现多条轨迹
    \end{itemize}
\end{frame}

\begin{frame}{眼图征象 5/5：反射类——幅度上几乎隐形}
    \centering
    \includegraphics[width=0.8\textwidth]{pic/eye_grp_impedance.png}
    \begin{itemize}\footnotesize
        \item 3 段电感 $\times z$ $\Rightarrow$ 特征阻抗 $Z=\sqrt{L'/C'}$ 出现台阶，能量进入\hl{反射}而非衰减
        \item 眼高几乎不变（354 vs 352/407 mV）$\Rightarrow$ 任何\hl{单标量裕度完全看不见它}；征象只藏在\hl{边沿附近的反射纹波}里——这正是后面区域消融中``平台区准确率 0.03''的伏笔
    \end{itemize}
\end{frame}

\section{结论与展望}

\begin{frame}{局限}
    \begin{itemize}
        \item 仿真数据，物理参数有文献依据但\hl{未经实测硅校准}
        \item 接收端的简单建模：不含 CTLE/DFE 均衡与 CDR
        \item 运行时的真实观测是误码序列/裕度计数
        \item 多标签报警的健康误报偏高 $\Rightarrow$ 需二分类健康门把守
        \item 开集识别揭示了故障之间的等价性
    \end{itemize}
\end{frame}

\begin{frame}{结论：三句话}
    \begin{enumerate}
        \item 随机工作负载下，\hl{从原始信号学习表征是必需}：线性模型 0.221 $\to$ CNN 0.789，且结构要\hl{匹配信号生成机制}
        \item \hl{遥测的价值取决于观测的有损性}：完整波形下 +1.4 pt；有损标量下高应力误报 49\%$\to$9\%、零误报检出 46.5\%$\to$61.1\%
        \item 全部基于开源 ngspice + Python，6000 样本 20 分钟可复现——一个\hl{可证伪}的 D2D 在役诊断基准
    \end{enumerate}
